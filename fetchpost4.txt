const data = {
    "email": "peter@klaven"
} ;
fetch('https://reqres.in/api/login', {
  method: 'POST', // or 'PUT'
  headers: {  
    'Content-Type': 'application/json',
  },
  body: JSON.stringify(data),
})
.then((response) => response.json())
.then((data) => {
  console.log('Success:', data);
})
.catch((error) => {
  console.error('Error:', error);
});
